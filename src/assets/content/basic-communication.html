Send information from the components.ts to the component.html

In the compoents.ts create a verible with in the export class with some information



    import { Component } from '@angular/core';

    @Component({
        selector: 'app-comunication-basic',
        standalone: true,
        imports: [],
        templateUrl: './comunication-basic.component.html',
        styleUrl: './comunication-basic.component.css'
    })

    export class ComunicationBasicComponent {

        //Create a verible with information
        sampleText:string = "This is sample text";

    }

In the compoents.html read the verible



    {{ sampleText }}

Send information in an object from the components.ts using a custom Interface to the component.html



    import { Component } from '@angular/core';

    //Create the interface
    interface SampleInterface
    {
        id: string;
        title: string;
    }

    @Component({
        selector: 'app-comunication-basic',
        standalone: true,
        imports: [],
        templateUrl: './comunication-basic.component.html',
        styleUrl: './comunication-basic.component.css'
    })
    export class ComunicationBasicComponent {
    
    //Create an instance of the interface and populate with information
    sampleInterface: SampleInterface[] = [
        { 
            id: "100",
            title: "Title One"
        },
        {
            id: "200",
            title: "Title Two"
        }
    ]

    }

In the components.html read the veribles. There are two elements in the sampleInterface array, so we are printing both of them




 {{ sampleInterface[0].id }}


 {{ sampleInterface[0].title }}


 


 {{ sampleInterface[1].id }}


 {{ sampleInterface[1].title }}
